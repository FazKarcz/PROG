include <iostream>
#include <vector>
#include <map>
using namespace std;
void wypisz(int a)
{
    cout << a << "\n";
}
void wypisz(vector<int>tab)
{
    for (int i = 0; i < tab.size(); i++)
    {
        cout << tab[i];
    }
}
void wypisz(map<int, int> m)
{
    for (auto elem : m)
    {
        cout << elem.first << "\t" << elem.second << "\n";
    }
}
bool czyNalezy(vector<int>tab, int element)
{
    for (int i = 0; i < tab.size(); i++)
    {
        if (tab[i] == element)
        {
            return true;
        }
    }
    return false;
}
bool czyNalezyMAPA(map<int, int>m,int klucz)
{
    for (auto item: m)
    {
        if (item.first == klucz)
        {
            return true;
        }
    }
    return false;
}
bool czyZawiera(vector<int>tab, vector<int>tab2)
{
    for (int i = 0; i < tab.size(); i++)
    {
        if (!czyNalezy(tab, tab2[i]))
        {
            return false;
        }
    }
    return true;
}
int min(vector<int>tab)
{
    int k = 0;
    for (int i = 0; i < tab.size(); i++)
    {
        if (tab[i] < k)
        {
            k = tab[i];
        }
    }
    return k;
}
vector<int>unikalna(vector<int>tab)
{
    vector<int>wynik;
    {
        for (int i = 0; i < tab.size(); i++)
        {
            if (!czyNalezy(wynik, tab[i]))
            {
                wynik.push_back(tab[i]);
            }
        }
    }
    return wynik;
}
map<int, int>wystepowanie(vector<int>tab)
{
    map<int, int>wynik;
    for (int i = 0; i < tab.size(); i++)
    {
        if (!czyNalezyMAPA(wynik, tab[i]))
        {
            wynik.insert({ tab[i],1 });
        }
        else
        {
            wynik[tab[i]]++;
        }
    }
    return wynik;
}
int main()
{
    vector<int>tab = { 2,3,9,4,1 };
    vector<int>tab2 = { 1,1,2,2,3,3 };
    map<int, int>m;
    m.insert({ 3,1 });
    m.insert({ 2,5 });
    m.insert({ 3,9 });
    if (czyNalezyMAPA(m, 8))
    {
        std::cout << "Nalezy klucz\n";
    }
    else
    {
        std::cout << "Nie nalezy klucz\n";
    }
    wypisz(wystepowanie(tab2));
}
