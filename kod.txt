#include <iostream>
#include <vector>
using namespace std;

void wypisz(int x)
{
    cout << x << endl;
}

void wypisz(vector<int> tab)
{
    for (int i = 0; i < tab.size(); i++)
    {
        cout << tab[i] << " ";
    }
    cout << endl;
}

void wypisz(vector<vector<int>> tab)
{
    wypisz(tab[0]);
    wypisz(tab[1]);
}

bool czynalezy(vector<int> tab, int x)
{
    for (int i = 0; i < tab.size(); i++)
    {
        if (tab[i] == x)
        {
            return true;
        }
    }
    return false;
}

bool czyzawiera(vector<int> tab1, vector<int> tab2)
{
    for (int i = 0; i < tab2.size(); i++)
    {
        if (!czynalezy(tab1, tab2[i]))
            return false;
    }
    return true;
}

int suma(int a, int b)
{
   return a + b;
}

int max(vector<int> tab)
{
    int wynik = tab[0];
    for (int i = 0; i < tab.size(); i++)
    {
        if (wynik < tab[i])
            wynik = tab[i];
    }
    return wynik;
}

vector<int> unikalnosc(vector<int> tab)
{
    vector<int> wynik;
    for (int i = 0; i < tab.size(); i++)
    {
        if (!czynalezy(wynik, tab[i]))
            wynik.push_back(tab[i]);
    }
    return wynik;
}
vector<vector<int>> czestosc(vector<int> tab)
{
   vector<vector<int>> wynik;
    vector<int> u = unikalnosc(tab);
    vector<int> c;

    for (int i = 0; i < u.size(); i++)
    {
        int ile = 0;
        for (int j = 0; j < tab.size(); j++)
        {
            if (tab[j] == u[i])
                ile++;
        }
        c.push_back(ile);
    }
    wynik.push_back(u);
    wynik.push_back(c);
    return wynik;
}



int main()
{
    std::cout << "Hello World!\n";
    vector<int> tab1 = { 3,5,5,1,2,8,8 };
    vector<int> tab2 = { 5,5,1 };

    wypisz(max(tab1));
    wypisz(czynalezy(tab1, 5));
    wypisz(czyzawiera(tab1, tab2));
    wypisz(unikalnosc(tab1));
    wypisz(tab1);
    wypisz(czestosc(tab1));
}
