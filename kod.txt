#include "pch.h"
#include <iostream>
#include <vector>
#include <map>
using namespace std;
void wypisz(int a)
{
	cout << a << "\n";
}
void wypisz(vector<int>tab)
{
	for (int i = 0; i < tab.size(); i++)
	{
		cout << tab[i] << " ";
	}
	cout << "\n";
}
void wypisz(map<int, int>m)
{
	for (auto jd : m)
	{
		cout << jd.first << " " << jd.second << "\n";
	}
}
bool czyNalezy(vector<int>tab, int cos)
{
	for (int i = 0; i < tab.size(); i++)
	{
		if (tab[i] == cos)
		{
			return true;
		}
	}
	return false;
}
bool czyNalezyMAPA(map<int, int>m, int kluczens)
{
	for (auto jd : m)
	{
		if (jd.first == kluczens)
		{
			return true;
		}
	}
	return false;
}
bool czyZawiera(vector<int>tab, vector<int>tab2)
{
	for (int i = 0; i < tab.size(); i++)
	{
		if (!czyNalezy(tab, tab2[i]))
		{
			return false;
		}
	}
	return true;
}
int min(vector<int>tab)
{
	int a = tab[0];
	for (int i = 0; i < tab.size(); i++)
	{
		if (tab[i] < a)
		{
			tab[i] = a;
		}
	}
	return a;
}
vector<int>unikalna(vector<int>tab)
{
	vector<int>wynik;
	for (int i = 0; i < tab.size(); i++)
	{
		if (!czyNalezy(wynik, tab[i]))
		{
			wynik.push_back(tab[i]);
		}
	}
	return wynik;
}
map<int, int>wystepowanie(vector<int>tab)
{
	map<int, int>wynik;
	for (int i = 0; i < tab.size(); i++)
	{
		if (!czyNalezyMAPA(wynik, tab[i]))
		{
			wynik.insert({ tab[i],1 });
		}
		else
		{
			wynik[tab[i]];
		}
	}
	return wynik;
}
vector<int>dzielnik(vector<int>tab, int dupa,int reszta=0)
{
	vector<int>wynik;
	for (int i = 0; i < tab.size(); i++)
	{
		if (tab[i] % dupa == reszta)
		{
			wynik.push_back(tab[i]);
		}
	}
	return wynik;
}
vector<int>generuj(int jd, int min = 0, int max = 50)
{
	vector<int>wynik;
	for (int i = 0; i < jd; i++)
	{
		wynik.push_back(rand() % (max - min) + min);
	}
	return wynik;
}
