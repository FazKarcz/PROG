#include "pch.h"
#include <iostream>
#include <vector>
using namespace std;


void wypisz(int a)
{
	cout << a << "\n";
}


void wypisz(vector<int>tab)
{
	for (int i = 0; i < tab.size(); i++)
	{
		cout << tab[i] << " ";
	}
	cout << "\n";
}


bool czyNalezy(vector<int>tab, int element)
{
	for (int i = 0; i < tab.size(); i++)
	{
		if (tab[i] == element)
		{
			return true;
		}
	}
	return false;
}


bool czyZawiera(vector<int>tab, vector<int>tab2)
{
	for (int i = 0; i < tab.size(); i++)
	{
		if (!czyNalezy(tab, tab2[i]))
		{
			return false;
		}
	}
	return true;
}
int Min(vector<int>tab)
{
	int m = tab[0];
	for (int i = 0; i < tab.size(); i++)
	{
		if (tab[i] < m)
		{
			m = tab[i];
		}

	}
	return m;
}
int Max(vector<int>tab)
{
	int m = tab[0];
	for (int i = 0; i < tab.size(); i++)
	{
		if (tab[i] > m)
		{
			m = tab[i];
		}
	}
	return m;
}

int suma(int a, int b)
{
	return a + b;
}


int main()
{
	int a, b;
	a = 2;
	b = 3;

	wypisz(suma(a, b));
	vector<int>tab = { 1,3,7 };
	vector<int>tab2 = { 4,3,2 };
	if (czyNalezy(tab, 2))
	{
		cout << "masno";
	}
	else
	{
		cout << "mamale";
	}
	cout << "\n";
	if (czyZawiera(tab, tab2))
	{
		cout << "proste";
	}
	else
	{
		cout << "krzywe";
	}
	cout << "\n";
	wypisz(Min(tab2));
	wypisz(Max(tab));
}
